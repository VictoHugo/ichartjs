这是一个基本的3D饼状图例子，例子展示了2012年7月份中国浏览器市场占有率Top6。此例子中，配置了图例。设置图例位置为底部居中，并以一行显示图例。数据项以百分比的形式展示数据值。
备注： 数据来源于StatCounter。
var data = [ {name : 'Other',value : 3.79,color:'#bd355e'}, {name : 'IE',value : 52.20,color:'#a5c2d5'}, {name : '360 Browser',value : 26.55,color:'#cbab4f'}, {name : 'Chrome',value : 8.26,color:'#76a871'}, {name : 'Firefox',value : 4.06,color:'#9f7961'}, {name : 'Maxthon',value : 2.78,color:'#a56f8f'}, {name : 'Safari',value : 2.35,color:'#db9034'} ]; $(function(){ new iChart.Pie3D({ render : 'canvasDiv', data: data, title : '2012年7月份中国浏览器市场占有率Top6', footnote : 'Data from StatCounter', legend : { enable : true, align : 'center', valign : 'bottom', row:1, column:'max' }, showpercent:true, decimalsnum:2, yHeight:80, radius:180, width : 800, height : 500 }).draw(); });
