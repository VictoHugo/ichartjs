这是一个基本的3D饼状图例子，例子展示了2012年2月份市场浏览器占有的情况。此例子中，配置了图例。设置图例位置为底部居中，并以一行显示图例。数据项以百分比的形式展示数据值。
备注： 数据均为模拟，无实际参考价值。
var data = [ {name : 'IE',value : 35.75,color:'#a5c2d5'}, {name : 'Chrome',value : 29.84,color:'#cbab4f'}, {name : 'Firefox',value : 24.88,color:'#76a871'}, {name : 'Safari',value : 6.77,color:'#9f7961'}, {name : 'Opera',value : 2.02,color:'#a56f8f'}, {name : 'Other',value : 0.73,color:'#6f83a5'} ]; $(function(){ new iChart.Pie3D({ render : 'canvasDiv', data: data, title : 'Top 5 Browsers from 1 to 29 Feb 2012', legend : { enable : true, align : 'center', valign : 'bottom', row:1, column:'max' }, showpercent:true, decimalsnum:2, width : 800, height : 400, radius:160 }).draw(); });
