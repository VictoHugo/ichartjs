这是一个条形图例子，该例子展示了2012年7月份全球范围内搜索引擎使用的情况。
备注： 数据来源于StatCounter。
var data = [ {name : 'Google',value : 91.17,color:'#a5c2d5'}, {name : 'bing',value : 3.22,color:'#cbab4f'}, {name : 'Yahoo!',value : 2.95,color:'#76a871'}, {name : 'Babylon',value : 0.54,color:'#9f7961'}, {name : 'Baidu',value : 0.45,color:'#a56f8f'}, {name : 'Other',value : 1.67,color:'#6f83a5'} ]; $(function(){ new iChart.Bar2D({ render : 'canvasDiv', data: data, title : '2012年7月份全球搜索引擎Top5', footnote : 'Data from StatCounter', width : 800, height : 440, coordinate:{ width:640, height:320, scale:[{ position:'bottom', start_scale:0, end_scale:100, scale_space:10, listeners:{ parseText:function(t,x,y){ return {text:t+"%"} } } }] }, rectangle:{ listeners:{ drawLabelText:function(r,t){ return t+"%"; } } } }).draw(); });
