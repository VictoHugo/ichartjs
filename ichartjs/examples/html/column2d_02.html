这是一个柱形图例子，该例子展示了2012年2月份市场浏览器占有的情况。此例子中，配置了图例、提示以及个性化的值轴。并以百分比的形式展示数据值。
备注： 数据均为模拟，无实际参考价值。
var data = [ {name : 'IE',value : 35.75,color:'#a5c2d5'}, {name : 'Chrome',value : 29.84,color:'#cbab4f'}, {name : 'Firefox',value : 24.88,color:'#76a871'}, {name : 'Safari',value : 6.77,color:'#9f7961'}, {name : 'Opera',value : 2.02,color:'#a56f8f'}, {name : 'Other',value : 0.73,color:'#6f83a5'} ]; $(function(){ new iChart.Column2D({ render : 'canvasDiv', data: data, title : 'Top 5 Browsers from 1 to 29 Feb 2012', showpercent:true, decimalsnum:2, width : 800, height : 400, legend : { enable : true }, tip:{ enable : true }, coordinate:{ width:580, scale:[{ position:'left', start_scale:0, end_scale:40, scale_space:8, listeners:{ parseText:function(t,x,y){ return {text:t+"%"} } } }] } }).draw(); });
