在基本的区域图例子基础上，配置了提示、图例和个性化的值轴。而且改变了一下背景颜色。但是没有被指底端的值轴。
备注： 数据均为模拟，无实际参考价值。
var data = [ { name : '北京', value:[2,12,14,20,28,32,34,36,33,24,14,4], color:'#c0abf1' } ]; var labels = ["一月","二月","三月","四月","五月","六月", "七月","八月","九月","十月","十一月","十二月"]; $(function(){ new iChart.Area2D({ render : 'canvasDiv', data: data, title : 'Average Temperature Of 2012', width : 800, height : 400, legend : { enable : true }, tip:{ enable : true }, listeners:{ parseTipText:function(d,t,i){ return labels[i]+份平均温度:+t+"℃"; } }, coordinate:{ scale:[{ position:'left', start_scale:0, end_scale:40, scale_space:5, listeners:{ parseText:function(t,x,y){ return {text:t+"℃"} } } }] } }).draw(); });
